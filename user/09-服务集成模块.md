### OAuth 登录流程

#### 1. 获取登录 URL
```typescript
// server/services/feishu.ts
export const getFeishuLoginUrl = () => {
  const params = new URLSearchParams({
    app_id: process.env.FEISHU_APP_ID!,
    redirect_uri: process.env.FEISHU_REDIRECT_URI!,
    state: generateRandomState() // 防 CSRF
  })

  return `https://open.feishu.cn/open-apis/authen/v1/index?${params}`

```

#### 2. 处理回调
```typescript
export const handleFeishuCallback = async (code: string) => {
  // 1. 获取 access_token
  const tokenResponse = await fetch(
    'https://open.feishu.cn/open-apis/authen/v1/access_token',
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        grant_type: 'authorization_code',
        client_id: process.env.FEISHU_APP_ID,
        client_secret: process.env.FEISHU_APP_SECRET,
        code
      })
    }
  )

  const { access_token } = await tokenResponse.json()

  // 2. 获取用户信息
  const userInfo = await getFeishuUserInfo(access_token)

  return userInfo
}
```

#### 3. 获取用户信息
```typescript
export const getFeishuUserInfo = async (accessToken: string) => {
  const response = await fetch(
    'https://open.feishu.cn/open-apis/authen/v1/user_info',
    {
      headers: {
        'Authorization': `Bearer ${accessToken}`
      }
    }
  )

  const data = await response.json()

  return {
    feishu_user_id: data.data.user_id,
    name: data.data.name,
    avatar_url: data.data.avatar_url,
    email: data.data.email
  }
}
```

### 飞书 API 调用

#### Tenant Access Token
```typescript
let cachedTenantToken: {
  token: string
  expires_at: number
} | null = null

export const getTenantAccessToken = async () => {
  // 检查缓存
  if (cachedTenantToken && cachedTenantToken.expires_at > Date.now()) {
    return cachedTenantToken.token
  }

  // 获取新 token
  const response = await fetch(
    'https://open.feishu.cn/open-apis/auth/v3/tenant_access_token/internal',
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        app_id: process.env.FEISHU_APP_ID,
        app_secret: process.env.FEISHU_APP_SECRET
      })
    }
  )

  const data = await response.json()

  // 缓存 token（提前 5 分钟过期）
  cachedTenantToken = {
    token: data.tenant_access_token,
    expires_at: Date.now() + (data.expire - 300) * 1000
  }

  return cachedTenantToken.token
}
```

#### 调用飞书 API
```typescript
export const callFeishuAPI = async (
  endpoint: string,
  options: RequestInit = {}
) => {
  const token = await getTenantAccessToken()

  const response = await fetch(
    `https://open.feishu.cn/open-apis${endpoint}`,
    {
      ...options,
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json',
        ...options.headers
      }
    }
  )

  if (!response.ok) {
    throw new Error(`飞书 API 调用失败: ${response.statusText}`)
  }

  return response.json()
}
```

## 错误处理

### 飞书 API 错误
```typescript
export class FeishuAPIError extends Error {
  code: number

  constructor(message: string, code: number) {
    super(message)
    this.code = code
    this.name = 'FeishuAPIError'
  }
}

export const handleFeishuError = (error: any) => {
  if (error.code === 99991663) {
    throw new FeishuAPIError('Token 已过期', error.code)
  }

  if (error.code === 99991668) {
    throw new FeishuAPIError('应用权限不足', error.code)
  }

  throw new FeishuAPIError(error.msg || '未知错误', error.code)
}
```

### 重试机制
```typescript
export const retryFeishuAPI = async <T>(
  fn: () => Promise<T>,
  maxRetries = 3
): Promise<T> => {
  let lastError: Error

  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn()
    } catch (error) {
      lastError = error as Error

      // 如果是 token 过期，清除缓存重试
      if (error instanceof FeishuAPIError && error.code === 99991663) {
        cachedTenantToken = null
      }

      // 等待后重试
      await new Promise(resolve => setTimeout(resolve, 1000 * (i + 1)))
    }
  }

  throw lastError!
}
```

## Webhook 处理

### 飞书事件订阅
```typescript
import crypto from 'crypto'

// 验证飞书事件签名
export const verifyFeishuSignature = (
  timestamp: string,
  nonce: string,
  encryptKey: string,
  body: string,
  signature: string
): boolean => {
  const data = `${timestamp}${nonce}${encryptKey}${body}`
  const hash = crypto
    .createHash('sha256')
    .update(data)
    .digest('hex')

  return hash === signature
}

// 处理飞书事件
export const handleFeishuEvent = async (event: any) => {
  switch (event.type) {
    case 'url_verification':
      // 首次验证
      return { challenge: event.challenge }

    case 'event_callback':
      // 处理具体事件
      await processFeishuCallback(event.event)
      return { success: true }

    default:
      console.warn('未知的飞书事件类型:', event.type)
      return { success: false }
  }
}
```

## 使用示例

### 登录流程
```typescript
// 前端：获取登录 URL
const loginUrl = await api.get('/api/auth/feishu')
window.location.href = loginUrl

// 后端：处理回调
router.get('/api/auth/feishu/callback', async (req, res) => {
  const { code } = req.query

  try {
    const userInfo = await handleFeishuCallback(code as string)

    // 创建或更新用户
    let user = db.prepare(
      'SELECT * FROM users WHERE feishu_user_id = ?'
    ).get(userInfo.feishu_user_id)

    if (!user) {
      const result = db.prepare(`
        INSERT INTO users (name, feishu_user_id, avatar_url, email)
        VALUES (?, ?, ?, ?)
      `).run(
        userInfo.name,
        userInfo.feishu_user_id,
        userInfo.avatar_url,
        userInfo.email
      )

      user = { id: result.lastInsertRowid, ...userInfo, role: 'user' }
    }

    // 保存会话
    req.session.user = user

    res.redirect('/home')
  } catch (error) {
    res.redirect('/login?error=auth_failed')
  }
})
```

## 注意事项

1. **Token 安全**：
   - 不要在前端暴露 App Secret
   - Token 仅在服务端使用
   - 定期轮换密钥

2. **API 限流**：
   - 遵守飞书 API 调用频率限制
   - 实现请求队列
   - 缓存常用数据

3. **错误处理**：
   - 完善的错误日志
   - 友好的错误提示
   - 自动重试机制

4. **权限范围**：
   - 只申请必要的权限
   - 遵循最小权限原则
   - 定期审查权限使用
